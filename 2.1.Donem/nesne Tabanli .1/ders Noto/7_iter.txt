			التكرار

القوائم ، المجموعات ، القواميس والمجموعات كلها كائنات قابلة للتكرار. إنها حاويات متكررة يمكنك الحصول على مكرر منها.
كل هذه الكائنات لها طريقة iter () التي تُستخدم للحصول على مكرر:

1.
mytuple = ("apple", "banana", "cherry")
myit = iter(mytuple)
print(next(myit))
print(next(myit))
print(next(myit))

output=>	apple
	banana
	cherry
------------------------------------------------------------------------------------------------------------

3.
class MyNumbers:
  def __iter__(self):
    self.a = 1
    return self

  def __next__(self):
    x = self.a
    self.a += 1
    return x

myclass = MyNumbers()
myiter = iter(myclass)

print(next(myiter))
print(next(myiter))
print(next(myiter))
print(next(myiter))
print(next(myiter))
output=>	1
	2
	3
	4
	5
